<div className="mt-16 mb-16 max-w-5xl mx-auto px-6">
  <h2 className="text-2xl sm:text-3xl font-bold text-gray-900 dark:text-white mb-4">Quick start</h2>
  <p className="text-base text-gray-600 dark:text-gray-400 mb-8">Get up and running with Motion Start in minutes</p>
  
  <Steps>
    <Step title="Install Motion Start">
      Add Motion Start to your Svelte project using your preferred package manager:
      
      <CodeGroup>
      ```bash npm
      npm install motion-start
      ```
      ```bash pnpm
      pnpm add motion-start
      ```
      ```bash yarn
      yarn add motion-start
      ```
      ```bash bun
      bun add motion-start
      ```
      </CodeGroup>
    </Step>
    
    <Step title="Import the Motion component">
      Import Motion into your Svelte component:
      
      ```svelte
      <script>
        import { Motion } from 'motion-start'
      </script>
      ```
    </Step>
    
    <Step title="Add your first animation">
      Create an animated element with declarative props:
      
      ```svelte
      <Motion.div
        initial={{ opacity: 0, y: 20 }}
        animate={{ opacity: 1, y: 0 }}
        transition={{ duration: 0.5 }}
      >
        Hello, Motion Start!
      </Motion.div>
      ```
      
      <Info>
        Motion Start provides animated versions of all HTML and SVG elements via the `Motion` namespace (e.g., `Motion.div`, `Motion.svg`, `Motion.path`).
      </Info>
    </Step>
  </Steps>
</div>
